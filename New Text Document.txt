<script>
function showGrades() {
    const inputs = document.querySelectorAll('#gradesForm input[type="number"]');
    let result = "<strong>درجات الطلاب:</strong><br>";
    inputs.forEach(input => {
        const name = input.name;
        const grade = input.value || "لم يتم الإدخال";
        result += `${name} : ${grade}<br>`;
    });
    document.getElementById('result').innerHTML = result;
}
</script>

<script>
function showAbsence() {
    const checkboxes = document.querySelectorAll('input[name="student"]:checked');
    let names = [];
    checkboxes.forEach(cb => names.push(cb.value));

    const resultDiv = document.getElementById('result');
    if (names.length > 0) {
        resultDiv.innerHTML = "<strong>الطلاب الغائبين:</strong><br>" + names.join("<br>");
    } else {
        resultDiv.innerHTML = "<strong>لا يوجد طلاب غائبين</strong>";
    }
}
</script>
